}); updateView(); } function updateView() { summary.textContent = `Matching rows: ${filteredData.length}`; renderTable(); renderChart(); } function renderTable() { previewTable.innerHTML = ""; if (!filteredData.length) return; } function downloadCSV() { const csv = Papa.unparse(filteredData); const blob = new Blob([csv], { type: "text/csv;charset=utf-8;" }); const url = URL.createObjectURL(blob); const link = document.createElement("a");
